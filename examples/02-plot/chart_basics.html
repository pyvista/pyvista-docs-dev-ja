гғҒгғЈгғјгғҲгҒ®еҹәжң¬#

гҒ“гҒ®дҫӢгҒ§гҒҜпјҢгҒ•гҒҫгҒ–гҒҫгҒӘгӮҝгӮӨгғ—гҒ®гғҒгғЈгғјгғҲгӮ’гӮ·гғјгғігҒ«иҝҪеҠ гҒҷгӮӢж–№жі•гӮ’зӨәгҒ—гҒҰгҒ„гҒҫгҒҷпјҺгӮҲгӮҠиӨҮйӣ‘гҒӘдҫӢгҒЁгҒ—гҒҰпјҢеҗҢгҒҳгғ¬гғігғҖгғ©гғјгҒ§иӨҮж•°гҒ®гғҒгғЈгғјгғҲгӮ’гӮӘгғјгғҗгғјгғ¬гӮӨгҒЁгҒ—гҒҰзө„гҒҝеҗҲгӮҸгҒӣгӮӢж–№жі•гҒҜпјҢ гғҒгғЈгғјгғҲгҒ®гӮӘгғјгғҗгғјгғ¬гӮӨ гҒ«гҒӮгӮҠгҒҫгҒҷпјҺ

import numpy as np

import pyvista as pv

rng = np.random.default_rng(1)  # Seeded random number generator for consistent data generation

гҒ“гҒ®дҫӢгҒ§гҒҜпјҢгғ©гғігғҖгғ гҒ«жҠҪеҮәгҒ•гӮҢгҒҹ100еҖӢгҒ®гғҮгғјгӮҝгғқгӮӨгғігғҲгҒӢгӮү2ж¬Ўе…ғгҒ®ж•ЈеёғеӣігӮ’дҪңжҲҗгҒҷгӮӢж–№жі•гӮ’зӨәгҒ—гҒҫгҒҷпјҺгғҮгғ•гӮ©гғ«гғҲгҒ§гҒҜпјҢгғҒгғЈгғјгғҲгҒҜи»ёгӮ’иҮӘеӢ•зҡ„гҒ«гғӘгӮ№гӮұгғјгғ«гҒ—пјҢгғ—гғӯгғғгғҲгҒ•гӮҢгҒҹгҒҷгҒ№гҒҰгҒ®гғҮгғјгӮҝгҒҢиҰӢгҒҲгӮӢгӮҲгҒҶгҒ«гҒӘгҒЈгҒҰгҒ„гҒҫгҒҷпјҺгғҒгғЈгғјгғҲгҒ®дёҠгҒ§еҸігӮҜгғӘгғғгӮҜгҒҷгӮӢгҒЁпјҢгғҒгғЈгғјгғҲгҒ®гӮәгғјгғ гӮ„гғ‘гғігҒҢеҸҜиғҪгҒ§гҒҷпјҺ

x = rng.standard_normal(100)
y = rng.standard_normal(100)
chart = pv.Chart2D()
chart.scatter(x, y, size=10, style="+")
chart.show()

гғҮгғјгӮҝгғқгӮӨгғігғҲгӮ’з·ҡгҒ§зөҗгҒ¶гҒ«гҒҜпјҢд»ҘдёӢгҒ®дҫӢгҒ®гӮҲгҒҶгҒ«2DгҒ®гғ©гӮӨгғігғ—гғӯгғғгғҲгӮ’дҪңжҲҗгҒҷгӮӢгҒ“гҒЁгҒҢгҒ§гҒҚгҒҫгҒҷпјҺгҒҫгҒҹпјҢгӮ«гӮ№гӮҝгғ и»ёгҒ®зҜ„еӣІгӮ’иҮӘеҲҶгҒ§жҢҮе®ҡгҒҷгӮӢгҒ“гҒЁгҒ§пјҢгғ—гғӯгғғгғҲгҒ•гӮҢгҒҹгғҮгғјгӮҝгӮ’еӢ•зҡ„гҒ« 'гӮәгғјгғ гӮӨгғі' гҒҷгӮӢгҒ“гҒЁгҒҢгҒ§гҒҚгҒҫгҒҷпјҺ

x = np.linspace(0, 10, 1000)
y = np.sin(x**2)

import matplotlib.pyplot as plt

# First, create the matplotlib figure
f, ax = plt.subplots(
    tight_layout=True
)  # Tight layout to keep axis labels visible on smaller figures
alphas = [0.5 + i for i in range(5)]
betas = [*reversed(alphas)]
N = int(1e4)
data = [rng.beta(alpha, beta, N) for alpha, beta in zip(alphas, betas)]
labels = [f"$\\alpha={alpha:.1f}\\,;\\,\\beta={beta:.1f}$" for alpha, beta in zip(alphas, betas)]
ax.violinplot(data)
ax.set_xticks(np.arange(1, 1 + len(labels)))
ax.set_xticklabels(labels)
ax.set_title("$B(\\alpha, \\beta)$")

# Next, embed the figure into a pyvista plotting window
p = pv.Plotter()
chart = pv.ChartMPL(f)
chart.background_color = 'w'
p.add_chart(chart)
p.show()
